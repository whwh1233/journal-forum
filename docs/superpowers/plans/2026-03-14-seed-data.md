# Seed 数据生成脚本实施计划

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 创建 seed 脚本，为本地开发库灌入 50 用户 + 200 评论 + 100 帖子 + 300 帖子评论 + 80 投稿 + 互动数据 + 5 公告的中文仿真数据。

**Architecture:** 一个主入口 `seed.js` 按 FK 依赖顺序调用 6 个数据生成模块，每个模块导出 `seed()` 和 `reset()` 函数。所有数据通过 Sequelize 模型写入，确保与正常 API 操作产生的数据一致。

**Tech Stack:** Node.js, Sequelize ORM, @faker-js/faker (zh_CN locale), bcryptjs

**Spec:** `docs/superpowers/specs/2026-03-14-seed-data-design.md`

```js
// backend/scripts/seed.js
require('dotenv').config();
const { connectDB } = require('../config/database');
const { syncDatabase } = require('../models');

const seedUsers = require('./seedData/users');
const seedComments = require('./seedData/comments');
const seedPosts = require('./seedData/posts');
const seedSubmissions = require('./seedData/submissions');
const seedInteractions = require('./seedData/interactions');
const seedAnnouncements = require('./seedData/announcements');

const isReset = process.argv.includes('--reset');

const modules = [
  seedUsers,
  seedComments,
  seedPosts,
  seedSubmissions,
  seedInteractions,
  seedAnnouncements,
];

async function main() {
  await connectDB();
  await syncDatabase({ alter: false });

  if (isReset) {
    console.log('\n🗑️  Resetting seed data...\n');
    // 反向顺序清理（尊重 FK 依赖）
    for (const mod of [...modules].reverse()) {
      await mod.reset();
    }
    console.log('\n✅ Seed data reset complete.\n');
  } else {
    // 幂等检查
    const { User } = require('../models');
    const exists = await User.findOne({ where: { email: 'seed-0@test.com' } });
    if (exists) {
      console.error('\n❌ Seed data already exists. Run `npm run seed:reset` first.\n');
      process.exit(1);
    }

    console.log('\n🌱 Seeding data...\n');
    for (const mod of modules) {
      await mod.seed();
    }
    console.log('\n✅ Seed complete.\n');
  }

  process.exit(0);
}

main().catch(err => {
  console.error('Seed failed:', err);
  process.exit(1);
});
```

## Task 2: 用户模块 (users.js)

**Files:**
- Create: `backend/scripts/seedData/users.js`

**关键约束:**
- User.id 是 UUID char(36)，由 Sequelize defaultValue: UUIDV4 自动生成
- email 必须唯一，格式 `seed-{i}@test.com`
- password 必须 bcrypt hash（与 authController.register 一致）
- role: 48 user + 1 admin (seed-48) + 1 superadmin (seed-49)

- [ ] **Step 1: 实现 users.js**

```js
const { faker } = require('@faker-js/faker/locale/zh_CN');
const bcrypt = require('bcryptjs');
const { User } = require('../../models');
const { Op } = require('sequelize');

const SEED_COUNT = 50;
const SEED_EMAIL_PATTERN = 'seed-%@test.com';
const PASSWORD = 'SeedPass123!';

const INSTITUTIONS = [
  '北京大学', '清华大学', '复旦大学', '上海交通大学', '浙江大学',
  '南京大学', '中国科学技术大学', '武汉大学', '华中科技大学', '中山大学',
  '四川大学', '哈尔滨工业大学', '西安交通大学', '同济大学', '南开大学',
  '天津大学', '北京师范大学', '中国人民大学', '吉林大学', '山东大学',
];

async function seed() {
  console.log('  Seeding users...');
  const hash = await bcrypt.hash(PASSWORD, 10);
  const users = [];

  for (let i = 0; i < SEED_COUNT; i++) {
    let role = 'user';
    if (i === SEED_COUNT - 2) role = 'admin';
    if (i === SEED_COUNT - 1) role = 'superadmin';

    users.push({
      email: `seed-${i}@test.com`,
      password: hash,
      name: faker.person.fullName(),
      bio: faker.lorem.sentence(),
      institution: faker.helpers.arrayElement(INSTITUTIONS),
      role,
      status: 'active',
    });
  }

  await User.bulkCreate(users);
  console.log(`  ✅ Created ${SEED_COUNT} users (seed-0 ~ seed-${SEED_COUNT - 1}@test.com, password: ${PASSWORD})`);
}

async function reset() {
  console.log('  Resetting users...');
  const count = await User.destroy({
    where: { email: { [Op.like]: 'seed-%@test.com' } },
    force: true,
  });
  console.log(`  ✅ Deleted ${count} seed users`);
}

// Helper: 获取所有 seed 用户 ID
async function getSeedUserIds() {
  const users = await User.findAll({
    where: { email: { [Op.like]: 'seed-%@test.com' } },
    attributes: ['id'],
  });
  return users.map(u => u.id);
}

module.exports = { seed, reset, getSeedUserIds, SEED_EMAIL_PATTERN };
```

## Task 3: 期刊评论模块 (comments.js)

**Files:**
- Create: `backend/scripts/seedData/comments.js`

**关键约束:**
- 顶级评论（parentId=null）必须带 dimensionRatings（至少含 overallExperience）
- rating 字段 = dimensionRatings.overallExperience
- journalId 是 STRING(50)，从 DB 查已有期刊

- [ ] **Step 1: 实现 comments.js**

核心逻辑：
1. 查询 DB 中已有期刊 journalId 列表
2. 生成 ~140 条顶级评论（带 dimensionRatings）
3. 生成 ~60 条回复评论（parentId 指向已创建的顶级评论）
4. 对每个被评论的期刊调用 updateJournalRatingCache

- [ ] **Step 2: 实现 updateJournalRatingCache 逻辑**

从 `commentController.js` 中提取 `updateJournalRatingCache` 函数的逻辑，在 seed 脚本中复用。推荐直接 require commentController 或将该函数抽取到共享 utils。

最简单的方案：seed 完成后，对每个有评论的 journalId 执行一次缓存更新：

```js
const { Comment, JournalRatingCache } = require('../../models');
const DIMENSION_KEYS = ['reviewSpeed', 'editorAttitude', 'acceptDifficulty', 'reviewQuality', 'overallExperience'];

async function updateRatingCache(journalId) {
  const comments = await Comment.findAll({
    where: { journalId, parentId: null, isDeleted: false }
  });
  if (comments.length === 0) {
    await JournalRatingCache.destroy({ where: { journalId } });
    return;
  }
  const sums = {}, counts = {};
  for (const key of DIMENSION_KEYS) { sums[key] = 0; counts[key] = 0; }
  for (const c of comments) {
    if (c.dimensionRatings) {
      for (const key of DIMENSION_KEYS) {
        if (c.dimensionRatings[key] != null) {
          sums[key] += c.dimensionRatings[key];
          counts[key]++;
        }
      }
    }
  }
  const averages = {};
  for (const key of DIMENSION_KEYS) {
    averages[key] = counts[key] > 0 ? Math.round((sums[key] / counts[key]) * 10) / 10 : null;
  }
  const validAvgs = Object.values(averages).filter(v => v !== null);
  const rating = validAvgs.length > 0
    ? Math.round((validAvgs.reduce((a, b) => a + b, 0) / validAvgs.length) * 10) / 10
    : 0;
  await JournalRatingCache.upsert({
    journalId, rating, ratingCount: comments.length, ...averages
  });
}
```
